[GOAL]: detect_intent(player_message)

[TASK]:
INPUT [player_message: str]
SCHEMA [available_types: List[TaskType] FROM json.response_format]
CLASSIFY player_message -> List[IntentItem]
    IF player_message is gibberish
        SET global_confidence = 0.0
OUTPUT {[{ task_type: TaskType, confidence: float [0..1] }], global_confidence: float [0..1], reasoning: str }
CONSTRAINT len(intents) >= 1

[RULES]:
ASSERT output_format: JSON ONLY
REJECT: markdown, explanations, preamble
ASSERT task_type IN available_types
ASSERT confidence IN float [0..1] FOR_EACH IntentItem
ASSERT len(intents) >= 1
ASSERT global_confidence = AVG([item.confidence FOR item IN intents])
ASSERT reasoning LANGUAGE = {{language}}

[EXAMPLE_SHAPE]:
{"intent_detection":{"intents":[{"task_type":"scene_narration_render","confidence":0.9}],"global_confidence":0.9,"reasoning":"player asks to continue the story"}}
